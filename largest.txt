int largest(int *arr,int n) //O(n)
{
    int m = 0,i;
    for(i=0;i<n;i++) {
        if(arr[i]>arr[m]) m=i;
    }
    
    return m;
}

int idx(int *a,int n)//O(n^2)
{
    for(int i=0;i<n;i++)
    {
        bool flag = true;
        for(int j=0;j<n;j++)
        {
            if(a[j]>a[i]) {
                flag = false;
                break;
            }
        }
        
        if(flag) return i;
    }
    return -1;
}
