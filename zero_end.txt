#include <bits/stdc++.h>
using namespace std;

void swap(int &a,int &b)
{
    int t=a;
    a=b;
    b=t;
}

void o_mov0(int *a,int n) //O(n)
{
    int c=0;
    for(int i=0;i<n;i++)
    {
        if(a[i]!=0)
        {
            swap(a[i],a[c]);
            c++;
        }
    }
}

void mov0(int *a,int n) //O(n^2)
{
   for(int i=0;i<n;i++)
   {
       if(a[i]==0)
       {
           for(int j=i+1;j<n;j++)
           {
               if(a[j]!=0) {
                   swap(a[i],a[j]);
                   break;}
           }
       }
   }
}
void prt(int *a,int n)
{
    for(int i=0;i<n;i++) cout<<a[i]<<" ";
}
