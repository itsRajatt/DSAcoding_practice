using namespace std;

bool search(bool *a,int key)
{
	if(a[key-1]==1) return true;
	return false;
}

void insert(bool *a,int val)
{
	a[val-1]=1;
}

void del(bool *a,int k)
{
	a[k-1]=false;
}
