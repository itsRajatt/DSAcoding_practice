#include <bits/stdc++.h>
using namespace std;

int evenOdd(int *a, int n) //O(n*n)
{
    int res=0;
   for(int i=0;i<n;i++)
   {
       int c=1;
           for(int j=i+1;j<n;j++)
           {
               if((a[j]%2==0 && a[j-1]%2!=0)||(a[j]%2!=0 && a[j-1]%2==0))
               {
                   c++;
               }
               else break;
       }
       res=max(res,c);
   }
   return res;
}

int eveOdd(int *a,int n)//O(n) kadane used
{
    int res=1,c=1;
    for(int i=1;i<n;i++)
    {
        if((a[i]%2==0 && a[i-1]%2!=0)||(a[i]%2!=0 && a[i-1]%2==0))  
        {
            c++;
            res=max(res,c);
        }
        else c=1;
    }
    return res;
}
