int max1(int a[],int n) //O(n*n)
{
    int res=0;
    for(int i=0;i<n;i++)
    {
        int c=0;
        for(int j=i;j<n;j++)
        {
            if(a[j]==1) c++;
           else break;
        }
        res=max(res,c);
    }
    return res;
}
