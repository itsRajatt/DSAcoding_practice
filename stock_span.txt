#include <bits/stdc++.h>
using namespace std;

int maxProfit(int *price,int start, int end) //naive
{
    if(end<=start) return 0;
    int profit=0;
    
    for(int i=start;i<end;i++)
    {
        for(int j=i+1;j<=end;j++)
        {
            if(price[j]>price[i]) {
                int cur_profit= price[j]-price[i]+maxProfit(price,start,i-1)+maxProfit(price,j+1,end);
                profit=max(profit,cur_profit);
            }
        }
    }
    return profit;
}

int maxP(int *price,int n)//better
{
    int profit =0;
    for(int i=0;i<n-1;i++) if(price[i+1]>price[i]) profit+=(price[i+1]-price[i]);
    
    return profit;
}
