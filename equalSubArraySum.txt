int splitArray(int *a,int n)
{
    int leftSum=0,rightSum=0;
    for(int i=0;i<n;i++) leftSum+=a[i];
    
    for(int i=n-1;i>=0;i--)
    {
        rightSum+=a[i];
        leftSum-=a[i];
        
        if(leftSum==rightSum) return i;
    }
    return -1;
}

int printSplit(int *a,int n)
{
    int split=splitArray(a,n);
    
    for(int i=0;i<n;i++)
    {
        if(i==split) cout<<endl;
        cout<<a[i]<<" ";
    }
}
