#include <bits/stdc++.h>
using namespace std;

void swap(char &a,char &b)
{
    char c=a;
    a=b;
    b=c;
}

void permute(string s,int i=0)
{
    if(i==s.length()-1){
        cout<<s<<endl;
        return;
    }
    
    for(int j=i;j<s.length();j++)
    {
        swap(s[i],s[j]);
        permute(s,i+1);
        swap(s[j],s[i]);
    }
    
}

    permute("ABC");

    return 0;
