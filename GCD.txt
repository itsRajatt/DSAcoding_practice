#include<bits/stdc++.h>
using namespace std;

int euc(int a,int b)
{
    while(a!=b)
    {
        if(a>b) a-=b;
        else b-=a;
    }
    return a;
}

int opEuc(int a,int b)
{
    if(!b) return a;
    else return opEuc(b,a%b);
} 

int gcd1(int a , int b)
{
    int gc = min(a,b);
    while(gc>0)
    {
        if(a%gc==0 && b%gc==0)
        {
            return gc;
        }
        gc--;
    }
}
