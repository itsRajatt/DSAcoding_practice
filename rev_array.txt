#include <bits/stdc++.h>
using namespace std;

void swap(int &a,int &b)
{
    int t=a;
    a=b;
    b=t;
}

void rev(int *a,int n)
{
   int l=0,h=n-1;
   while(l<h) {
       swap(a[l],a[h]);
       l++;
       h--;
   }
   for(int i=0;i<n;i++) cout<<a[i]<<" ";
}
