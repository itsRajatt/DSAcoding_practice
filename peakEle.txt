#include<bits/stdc++.h>
using namespace std;

int peak_ele(int *a,int n)
{
    if(n==1) return a[0];
    if(a[0]>=a[1]) return a[0];
    if(a[n-1]>a[n-2]) return a[n-1];
    for(int i=1;i<n;i++)
    {
        if(a[i-1]<a[i]&& a[i]>a[i+1]) return a[i];
    }
}

int binSPeak(int *a , int n)
{
    int l=0,h=n-1;
    while(l<=h)
    {
        int mid=(l+h)/2;
        if((mid==0 || a[mid-1]<=a[mid])&&(mid==n-1||a[mid+1]<=a[mid])) return mid;
        if(mid>0 && a[mid-1]>=a[mid]) h=mid-1;
        else l=mid+1;
    }
    return -1;
}

    cout<<binSPeak(a,6);
