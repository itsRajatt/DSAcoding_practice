Node *middleLL(Node *head)
{
     Node *p=head;
    Node *q=head;
    while(q && q->next)
    {
        p=p->next;                 //to increase by 1
        q=q->next->next;           //to increase by 2
    }
    return p;
    
}
}

void printLL(Node *head)
{
    if(!head) {cout<<"-1"<<" "; return ;}
    cout<<head->data<<" ";
    head=head->next;
}

int main() {
    Node *head=new Node(5);
    head->next=new Node(10);
    head->next->next=new Node(25);
    head=middleLL(head);
    printLL(head);

    return 0;
