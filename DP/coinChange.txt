#include<bits/stdc++.h>
using namespace std;


int recTot_Sum(int coins[] , int n, int sum)
{
    if(!sum) return 1;
    if(!n) return 0;
    
    int res=recTot_Sum(coins,n-1,sum);
    if(coins[n-1]<=sum) res+=recTot_Sum(coins,n,sum-coins[n-1]);
    
    return res;
}

//DP -- as 2 param change n and sum , so take 2d array
int getCount(int coins[], int n, int sum)
{
    int dp[sum+1][n+1];
    
    for(int i=0;i<=n;i++)
    {
        dp[0][i]=1;
    }
    
    for(int j=0;j<=sum;j++)
    {
        dp[j][0]=0;
    }
    
    for(int i=1;i<=sum;i++)
    {
        for(int j=1;j<=n;j++)
        {
            dp[i][j] = dp[i][j-1];
            
            if(coins[j-1]<=i)
            dp[i][j]+=dp[i-coins[j-1]][j];
        }
    }
    
    return dp[sum][n];
    
}
