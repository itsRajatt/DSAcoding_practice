// Online C++ compiler to run C++ program online
#include <bits/stdc++.h>
using namespace std;

int sqrt(int x)
{
    int l=0,h=x,ans=-1;
    while(l<=h)
    {
        int mid = (l+h)/2;
        int mSq=mid*mid;
        
        if(mSq==x) return mid;
        else if(mSq>x) {
            h=mid-1;
        }
        else {
            l=mid+1;
            ans=mid;
        }
    }
    return ans;
}
