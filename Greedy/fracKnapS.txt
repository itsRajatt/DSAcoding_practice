/* 
O(nlogn+n)= O(nlogn)
first = val , second = weight

if weight of sorted item first <= weight of knapS , choose full item ; incre res+=arr[i].first and decrement weight of knapsack
otherwise add a fraction of the maxx val/weight res+=arr[i].first * ((double)w/arr[i].second) 

*/

bool myCmp(pair<int,int>a ,pair<int,int>b)
{
    double r1=(double)a.first/a.second;
    double r2=(double)b.first/b.second;
    
    return r1>r2;
}

double fknapS(int w , pair<int,int>arr[],int n)
{
    sort(arr,arr+n,myCmp);
    double res=0.0;
    for(int i=0;i<n;i++)
    {
        if(arr[i].second <= w)
        {
            res+=arr[i].first;
            w-=arr[i].second;
        }
        else 
        {
            res+=arr[i].first * ((double)w/arr[i].second);
            break;
        }
    }
    
    return res;
}


int main() {
    // Write C++ code here
    pair <int, int> arr[] = {make_pair(120, 30), make_pair(100, 20), make_pair(60, 10)};

	int n = 3, w = 50;

	cout<<fknapS(w, arr, n); 

	return 0;
   
}
