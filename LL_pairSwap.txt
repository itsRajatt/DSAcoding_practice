#include<bits/stdc++.h>
using namespace std;

struct Node
{
    int data;
    Node *next;
    
    Node(int x)
    {
        data=x;
       next=nullptr
    }
};

Node *pairwiseSwap(Node *head)
{
    //naive --> swap data and keep moving
    
    Node *t=head;
    
    while(t->next && t)
    {
        int c;
        c=t->data;
        t->data=t->next->data;
        t->next->data=c;
        
        t=t->next->next;
    }
    return head;
    
}

Node *pairSwap(Node *head) //change pointers (better)
{
    if(!head && !head->next) return head;
    Node *curr = head->next->next;
    Node *prev = head;
    head=head->next;
    head->next=prev;
    while(curr && curr->next)
    {
        prev->next=curr->next;
        prev=curr;
        Node *next=curr->next->next;
        curr->next->next=curr;
        curr=next;
    }
    prev->next=curr;
    return head;
}
