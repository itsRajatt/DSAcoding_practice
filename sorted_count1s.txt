// Online C++ compiler to run C++ program online
#include <bits/stdc++.h>
using namespace std;

int count(int *a ,int n)
{
    int l=0,h=n-1;
    
    while(l<=h)
    {
        int mid=(l+h)/2;
        if(a[mid]==0) l=mid+1;
        else {
            if(a[mid]==0 || a[mid-1]==0) return n-mid;
            else h=mid-1;
    }
}
    return 0;
}

    int a[]={0,0,1,1,1,1,1,1};
    cout<<count(a,8);

    return 0;
}
