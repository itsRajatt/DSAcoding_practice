#include <bits/stdc++.h>
using namespace std;

int SOD(int n) //recursive
{
    if(n<=9) return n;
    
    return (n%10)+SOD(n/10);
}

int SOD1(int n) //iterative
{
    int res=0;
    while(n>0)
    {
        res+=(n%10);
        n/=10;
    }
    return res;
}
