#include<bits/stdc++.h>
using namespace std;

bool sumTwoPtr(int *a,int x,int n)
{
    for(int i=0;i<n-1;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if(a[i]+a[j]==x) return 1;
        }
    }
    return 0;
}
//2ptr best for sorted array
//for unsorted use hashing
bool OsumTwoPtr(int *a,int n,int x) //O(n)
{
    int l=0,h=n-1;
    
    while(l<h)
    {
        if(a[l]+a[h]==x) return 1;
        else if(a[l]+a[h]<x) l++;
        else  h--;
    }
    return 0;
}

int main()
{
    int a[]={1,2,3,4,-1,11};
    cout<<OsumTwoPtr(a,6,16);
    
    
    return 0;
    
}
