#include<bits/stdc++.h>
using namespace std;

int lcm(int a,int b)
{
    int res = max(a,b);
    while(1)
    {
        if(res%a==0 && res%b==0) return res;
        res++;
    }
}

int gcd(int a,int b)
{
    if(b==0) return a;
    return gcd(b,a%b);
}

int opLcm(int a,int b)
{
    return (a*b)/(gcd(a,b));
}

int main()
{
    cout<<opLcm(10,25);
    return 0;
}
