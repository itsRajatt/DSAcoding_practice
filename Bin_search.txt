#include <bits/stdc++.h>
using namespace std;

int bs(int *a,int n,int k)
{
    int l=0;
    int h=n-1;
   
    while(l<=h)
    {
         int mid=(l+h)/2;
        if(a[mid]==k) return mid;
        else if(a[mid]>k) {
            h=mid-1;
        }
        else if(a[mid]<k){
            l=mid+1;
        }
    }
    return -1;
}

int rbs(int *a,int l,int h,int k)
{
    
    int mid =(l+h)/2;
    if(l>=h) return -1;
    if(a[mid]==k) return mid;
    else if(a[mid]>k) rbs(a,0,mid-1,k);
    else rbs(a,mid+1,h,k);
}
