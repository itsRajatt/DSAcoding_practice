#include <iostream>
using namespace std;

void lead(int *a,int n) //O(n^2)
{
    for(int i=0;i<n;i++)
    {
        bool flag=0;
        for(int j=i+1;j<n;j++)
        {
            if(a[j]>=a[i]) {flag=1; break;}
        }
         if(flag==0) {cout<<a[i]<<" ";}
    }
    
}

void o_lead(int *a,int n) //for same order -> use aux array and print in rev
{
    int curr=a[n-1];
    cout<<curr<<" ";
    for(int i=n-2;i>=0;i--)
    {
        if(curr<a[i])
        {
            curr=a[i];
            cout<<a[i]<<" ";
        }
    }
}

    o_lead(a,6);
    
    
    return 0;
}
