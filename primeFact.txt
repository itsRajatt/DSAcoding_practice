bool isPrime(int n)
{
    if(n==2 || n==3) return 1;
    if(n%2==0 || n%3==0 || n==1) return 0;
    
    for(int i=5;i*i<=n;i=i+6)
    {
        if(n%i==0 || n%(i+2)==0) return 0;
    }
    return 1;
}

{
    for(int i=2;i<n;i++)
    {
        if(isPrime(i))
        {
            int x =i;

                cout<<i<<" ";
                x*=i;

        }
    }
}
