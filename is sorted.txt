bool is_sorted(int *a,int n)
{
   for(int i=0;i<n;i++)
   {
       for(int j=i+1;j<n;j++)
       {
           if(a[j]<a[i]) return 0;
       }
   }
   return 1;
}

bool ois_sorted(int *a,int n)
{
   for(int i=1;i<n;i++)
   {
       if(a[i]<a[i-1]) return 0;
   }
   return 1;
}
