int minCoins(int coin[], int n, int amount)
{	
    	int res=0;
    	sort(coin,coin+n,greater<int>());
    	
    	for(int i=0;i<n;i++)
    	{
    	    int c =floor(amount/coin[i]);
    	    res+=c;
    	    amount-=c*coin[i];
    	    
    	    if(!amount) break;
    	}
    	return res;
}

      cout<<minCoins(coin, n, amount);
