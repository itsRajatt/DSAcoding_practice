#include <bits/stdc++.h>
using namespace std;

struct Node  
{ 
  int key; 
  struct Node *left; 
  struct Node *right; 
  Node(int k){
      key=k;
      left=right=NULL;
  }
};

bool isSum(Node *root) //O(n) & O(H)
{
    if(!root) return 1;
    if(!root->left && !root->right) return 1;
    
    int sum=0;
    if(root->left) sum+=root->left->key;
    if(root->right) sum+=root->right->key;
    
    return (root->key ==sum && isSum(root->left) && isSum(root->right));
}

int main() {
	
	Node *root=new Node(10);
	root->left=new Node(2);
	root->right=new Node(8);
	root->right->left=new Node(3);
	root->right->right=new Node(5);
	
	cout<<isSum(root);
}
