#include <bits/stdc++.h>
using namespace std;

struct Node  
{ 
  int key; 
  struct Node *left; 
  struct Node *right; 
  Node(int k){
      key=k;
      left=right=NULL;
  }
};

int sizeBT(Node *root) //can use iterative via LevelOrder , use q and loop while q.empty false . then inc the count  . Although AuxSpace differs
{
    if(!root) return 0;
    int sz = 1+sizeBT(root->left)+sizeBT(root->right);
    return sz;
}

int main() {
	
	Node *root=new Node(10);
	root->left=new Node(20);
	root->right=new Node(30);
	root->left->left=new Node(40);
	root->left->right=new Node(50);
	root->right->left=new Node(60);
	root->right->right=new Node(70);
	
	cout<<sizeBT(root);
}
