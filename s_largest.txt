// Online C++ compiler to run C++ program online
#include <bits/stdc++.h>
using namespace std;

int s_lar(int *a,int n)
{
    int res=-1,lar=0;
    for(int i=1;i<n;i++)
    {
        if(a[i]>a[lar]) 
        {
            res=lar;
            lar=i;
        }
        if(a[i]!=a[lar])
        {
            if(res==-1 || a[i]>a[res]) res=i;
        }
    }
    return res;
}

int main() {
    // Write C++ code here
    int a[]={1,4,2,60,9,123};
    std::cout << s_lar(a,6);

    return 0;
}
