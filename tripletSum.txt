#include<bits/stdc++.h>
using namespace std;

bool tripletSum(int *a,int n,int x)
{
    int l,h;
    
    for(int i=0;i<n-2;i++)
    {
        l=i+1,h=n-1;
        while(l<h)
        {
            if(a[i]+a[l]+a[h]==x) {
                cout<<a[i]<<" "<<a[l]<<" "<<a[h];
                return 1;
            }
            else if(a[i]+a[l]+a[h]<x) l++;
            else h--;
        }
    }
    return 0;
}

int main()
{
    int a[]={1,2,3,4,5,6};
    tripletSum(a,6,15);
}
