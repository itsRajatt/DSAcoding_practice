#include <bits/stdc++.h>

int dig(int n)
{
    int c=0;
    while(n>0)
    {
        n/=10;
        c++;
    }
    return c;
}

int dig1(int n)
{
    if(!n) return 0;
    
    return 1+dig1(n/10);
}

int dig2(int n)
{
    return floor(log10(n)+1);
}
